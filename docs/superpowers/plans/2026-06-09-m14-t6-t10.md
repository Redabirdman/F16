**Goal:** Finalize M14 — let operators edit any agent's prompt from the admin (generic registry + whole-prompt override, audited + reversible) and add a `/team-chat` page (operator timeline + send-to-WA-group).

**Architecture:** A central prompt registry (every prompt site registers a key + code default) + a `prompt_overrides` table; `resolvePrompt(db, key, getDefault)` returns the override else the default (safe fallback). **Tech Stack:** Hono, drizzle-orm, Postgres (pgvector), React + Vite + react-query + Tailwind, vitest. Spec: `docs/superpowers/specs/2026-06-09-m14-t6-t10-design.md`.

```ts
// backend/src/db/repositories/prompt-overrides.ts
import { eq, sql } from 'drizzle-orm';
import type { Database } from '../index.js';
import { promptOverrides, type PromptOverride } from '../schema/prompt-overrides.js';

export async function getOverride(db: Database, key: string): Promise<PromptOverride | null> {
  const [row] = await db
    .select()
    .from(promptOverrides)
    .where(eq(promptOverrides.key, key))
    .limit(1);
  return row ?? null;
}
export async function listOverrides(db: Database): Promise<PromptOverride[]> {
  return db.select().from(promptOverrides);
}
export async function upsertOverride(
  db: Database,
  key: string,
  content: string,
  updatedBy: string | null,
): Promise<void> {
  await db
    .insert(promptOverrides)
    .values({ key, content, updatedBy })
    .onConflictDoUpdate({
      target: promptOverrides.key,
      set: { content, updatedBy, updatedAt: sql`now()` },
    });
}
export async function deleteOverride(db: Database, key: string): Promise<boolean> {
  const res = await db
    .delete(promptOverrides)
    .where(eq(promptOverrides.key, key))
    .returning({ key: promptOverrides.key });
  return res.length > 0;
}
```

```ts
// backend/src/prompts/registry.ts
import type { Database } from '../db/index.js';
import { getOverride } from '../db/repositories/prompt-overrides.js';

export interface PromptDef {
  key: string;
  label: string;
  agentRole: string;
  description: string;
  getDefault: () => string;
}
const registry = new Map<string, PromptDef>();
let cacheVersion = 0;
const cache = new Map<string, { v: number; content: string | null }>();

export function registerPrompt(def: PromptDef): void {
  registry.set(def.key, def); // last registration wins (idempotent on hot reload)
}
export function listPromptDefs(): PromptDef[] {
  return [...registry.values()].sort(
    (a, b) => a.agentRole.localeCompare(b.agentRole) || a.key.localeCompare(b.key),
  );
}
export function getPromptDef(key: string): PromptDef | undefined {
  return registry.get(key);
}
export function bustPromptCache(): void {
  cacheVersion += 1;
}
/** Override if set, else the code default. NEVER throws — falls back to default. */
export async function resolvePrompt(
  db: Database,
  key: string,
  getDefault: () => string,
): Promise<string> {
  try {
    const c = cache.get(key);
    if (c && c.v === cacheVersion) return c.content ?? getDefault();
    const row = await getOverride(db, key);
    cache.set(key, { v: cacheVersion, content: row?.content ?? null });
    return row?.content ?? getDefault();
  } catch {
    return getDefault();
  }
}
export function __resetRegistryForTests(): void {
  registry.clear();
  cache.clear();
  cacheVersion = 0;
}
```

```ts
// backend/src/admin/prompts.ts
import { Hono } from 'hono';
import type { Database } from '../db/index.js';
import { listPromptDefs, getPromptDef, bustPromptCache } from '../prompts/registry.js';
import {
  listOverrides,
  upsertOverride,
  deleteOverride,
} from '../db/repositories/prompt-overrides.js';
import { appendAudit } from '../db/repositories/audit-log.js';

export function buildAdminPromptsRouter(opts: { db: Database }): Hono {
  const app = new Hono();

  app.get('/v1/admin/prompts', async (c) => {
    const defs = listPromptDefs();
    const overrides = new Map((await listOverrides(opts.db)).map((o) => [o.key, o]));
    return c.json({
      prompts: defs.map((d) => {
        const o = overrides.get(d.key);
        return {
          key: d.key,
          label: d.label,
          agentRole: d.agentRole,
          description: d.description,
          default: d.getDefault(),
          override: o?.content ?? null,
          isOverridden: Boolean(o),
          updatedAt: o ? o.updatedAt.toISOString() : null,
          updatedBy: o?.updatedBy ?? null,
        };
      }),
    });
  });

  app.put('/v1/admin/prompts/:key', async (c) => {
    const key = c.req.param('key');
    if (!getPromptDef(key)) return c.json({ error: 'unknown prompt key' }, 404);
    const body = (await c.req.json().catch(() => ({}))) as { content?: string };
    const content = (body.content ?? '').trim();
    if (!content) return c.json({ error: 'content required' }, 400);
    await upsertOverride(opts.db, key, content, 'admin');
    bustPromptCache();
    await appendAudit(opts.db, {
      actorType: 'human',
      actorId: 'admin',
      action: 'prompt.override.set',
      targetType: 'prompt',
      targetId: key,
      after: { length: content.length },
    });
    return c.json({ ok: true, key }, 200);
  });

  app.delete('/v1/admin/prompts/:key', async (c) => {
    const key = c.req.param('key');
    if (!getPromptDef(key)) return c.json({ error: 'unknown prompt key' }, 404);
    const removed = await deleteOverride(opts.db, key);
    bustPromptCache();
    await appendAudit(opts.db, {
      actorType: 'human',
      actorId: 'admin',
      action: 'prompt.override.reset',
      targetType: 'prompt',
      targetId: key,
      after: { removed },
    });
    return c.json({ ok: true, key, removed }, 200);
  });

  return app;
}
```

Mount in `index.ts` next to the other admin routers: `app.route('/', buildAdminPromptsRouter({ db: opts.db }));` + import.
